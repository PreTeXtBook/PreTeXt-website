Article =
    element article {
        MetaDataLinedSubtitle,
        ArticleFrontMatter?,
        (
            (
                Objectives?,
                (BlockDivision | Paragraphs)+,
                (ReadingQuestions? & Exercises? &
                 Solutions? & References? & Glossary?),
                Outcomes?
            )
        |
            (
                (Objectives? & IntroductionDivision?),
                Section,
                (Section | ReadingQuestions | Exercises |
                 Solutions | References | Glossary)*,
                (Outcomes? & ConclusionDivision?),
                ArticleBackMatter?
            )
        )
    }
